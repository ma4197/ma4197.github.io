Total Snowfall (in) By Year

samp_noaa %>% 
  filter(0 < snow) %>% 
  group_by(year) %>% 
  mutate(snow = snow*0.039371) %>% 
  summarize(snowed = sum(snow)) %>% 
  mutate(years = as.character(year)) %>% 
  plot_ly(x = ~years, y = ~snowed, color = ~snowed, type = "bar", colors = "viridis")
